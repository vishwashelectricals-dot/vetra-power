  if(editIdx === null){

    toast('Added: ' + name);
  } else {
    INV[editIdx] = { name:name, unit:unit, rate:rate };
    toast('Updated: ' + name);
    editIdx = null;
  }
  sortINV();
  saveInvStore();
  clearInvForm();
  renderInv();
  filterItems();
});

function renderBill(){
  var body = G('billbody');
  body.innerHTML = '';
  if(!BILL.length){
    body.innerHTML = '<tr><td colspan="7" style="text-align:center;color:#999;padding:18px;">No items added yet.</td></tr>';
    return;
  }
  BILL.forEach(function(item, i){
    var tot = (item.qty * item.rate).toFixed(2);
    var tr  = document.createElement('tr');
    tr.innerHTML =
      '<td style="text-align:center">' + (i+1) + '</td>' +
      '<td>' + item.name + '</td>' +
      '<td style="text-align:center"><input class="qin" type="number" value="' + item.qty + '" min="0.01" step="any" data-i="' + i + '"></td>' +
      '<td style="text-align:center">' + item.unit + '</td>' +
      '<td style="text-align:right">' + item.rate + '</td>' +
      '<td style="text-align:right;font-weight:bold">' + tot + '</td>' +
      '<td style="text-align:center" class="nopr"><button class="xbtn" data-i="' + i + '">X</button></td>';
    body.appendChild(tr);
  });

  // Attach qty change listeners
  body.querySelectorAll('.qin').forEach(function(inp){
    inp.addEventListener('change', function(){
      var idx = parseInt(inp.getAttribute('data-i'));
      var v   = parseFloat(inp.value);
      if(isNaN(v) || v <= 0){ v = 1; inp.value = 1; }
      BILL[idx].qty = v;
      saveBillStore();
      renderBill();
      calcTotals();
    });
  });

  // Attach delete listeners
  body.querySelectorAll('.xbtn').forEach(function(btn){
    btn.addEventListener('click', function(){
      var idx  = parseInt(btn.getAttribute('data-i'));
      var name = BILL[idx].name;
      BILL.splice(idx, 1);
      saveBillStore();
      renderBill();
      calcTotals();
      toast('Removed: ' + name);
    });
  });
}

// ============================================================
//  CALC TOTALS
// ============================================================
function calcTotals(){
  var sub = 0;
  BILL.forEach(function(it){ sub += it.qty * it.rate; });

  var dv = parseFloat(G('discV').value) || 0;
  var dt = G('discT').value;
  var tv = parseFloat(G('taxV').value)  || 0;
  var tt = G('taxT').value;

  var disc  = (dt === 'pct') ? (sub * dv / 100) : dv;
  var after = Math.max(0, sub - disc);
  var tax   = (tt === 'pct') ? (after * tv / 100) : tv;
  var grand = after + tax;

  G('tSub').textContent   = sub.toFixed(2);
  G('tGrand').textContent = grand.toFixed(2);

  if(dv > 0){
    G('tDisc').textContent    = disc.toFixed(2);
    G('drow').style.display   = 'flex';
  } else {
    G('drow').style.display   = 'none';
  }
  if(tv > 0){
    G('tTax').textContent     = tax.toFixed(2);
    G('txrow').style.display  = 'flex';
  } else {
    G('txrow').style.display  = 'none';
  }

  saveFields();
}

['discV','discT','taxV','taxT'].forEach(function(id){

  G(id).addEventListener('change', calcTotals);

  var mob   = G('pMobile').value.trim() || '';

  var grand = G('tGrand').textContent;
  var sub   = G('tSub').textContent;
  var rmk   = G('pRmk').value.trim();
  var dv    = parseFloat(G('discV').value) || 0;
  var tv    = parseFloat(G('taxV').value)  || 0;
  var disc  = G('tDisc').textContent;
  var tax   = G('tTax').textContent;

  var lines = [];
  lines.push('*VETRA POWER - Quotation*');
  lines.push('--------------------------------');
  lines.push('*Client:* ' + name);
  lines.push('*Quo No:* ' + qno);
  lines.push('*Date:* '   + date);
  lines.push('--------------------------------');
  BILL.forEach(function(it, idx){
    lines.push((idx+1) + '. ' + it.name);
    lines.push('   ' + it.qty + ' ' + it.unit + ' x Rs' + it.rate + ' = Rs' + (it.qty * it.rate).toFixed(2));
  });
  lines.push('--------------------------------');
  lines.push('Subtotal: Rs' + sub);
  if(dv > 0) lines.push('Discount: -Rs' + disc);
  if(tv > 0) lines.push('Tax/GST: +Rs' + tax);
  lines.push('*Grand Total: Rs' + grand + '*');
  if(rmk){ lines.push(''); lines.push('Note: ' + rmk); }
  lines.push('');
  lines.push('VETRA POWER | Deep Y. Shah');
  lines.push('Ph: 9537366551 / 9913587774');

  var msg = lines.join('\n');

  // Parse phone
  var phone = mob.replace(/\D/g, '');
  if(phone.length > 10){
    // remove country code
    phone = phone.slice(phone.length - 10);
  }

  var url;
  if(phone.length === 10){
    url = 'https://wa.me/91' + phone + '?text=' + encodeURIComponent(msg);
  } else {
    url = 'https://wa.me/?text=' + encodeURIComponent(msg);
  }
  window.open(url, '_blank');
}
